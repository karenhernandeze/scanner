void sort(float a[], int low, int high){
    int i; int k;
    i = low;

    while (i < high - 1){
        float t;
        k = miniloc(a,i,high);
        t = a[k];
        a[k] = a[i];
        a[i] = t;
        i = i +1;
    }
    return;
}/* END of sort() */
